233  inline double LineSearch(double const costVal, Eigen::VectorXd const& stepDir, Eigen::VectorXd& beta, Eigen::VectorXd& costVec) {
234  double alpha = 1.0;
235 
236  // evaluate the cost at the initial guess
237  EvaluateCost(beta - stepDir, costVec);

239  std::size_t iter = 0;
240  while( newCost>costVal & iter++<maxLineSearchSteps ) {
